#!/usr/bin/python3
"""
Test file for BaseGeometry class
"""

BaseGeometry = __import__('6-base_geometry').BaseGeometry

if __name__ == "__main__":
    bg = BaseGeometry()

    # Test area()
    try:
        bg.area()
    except Exception as e:
        print(e)

    print("-" * 30)

    # Test integer_validator with valid value
    try:
        bg.integer_validator("width", 5)
        print("width is valid")
    except Exception as e:
        print(e)

    # Test integer_validator with non-integer
    try:
        bg.integer_validator("width", "5")
    except Exception as e:
        print(e)

    # Test integer_validator with zero
    try:
        bg.integer_validator("height", 0)
    except Exception as e:
        print(e)

    # Test integer_validator with negative value
    try:
        bg.integer_validator("height", -3)
    except Exception as e:
        print(e)
